# List the files and folders in the folder:
$ ls

# Move to folder:
$ cd folder/
$ cd ..
$ cd <link_to_folder>

# Create a new folder:
$ mkdir name_folder

# Reduce the folder link
$ cd ~

# clear screen:
$ clear

# Delete folder:
$ rm -rf name_folder/

# Create a file:
$ touch file.txt
$ touch file.csv
$ touch file.xlsx

# Delete file:
$ rm file.txt
